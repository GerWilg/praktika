function[a] = unit_step(x)
for i = 1:numel(x)
    if x(i) < 0
        a(i) = 0;
    else 
        a(i) = 1;
    end
end
///////
function [a, t] = discrete (A, signal_f, fi0, d_f, t0, t1)
t = t0:1/d_f:t1;
for i = 1:numel(t)
    a(i) = A*sin(2*pi*signal_f*t(i)+fi0);
end
